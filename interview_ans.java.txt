import java.util.*;

class Song {
    private int id;
    private String name;

    public Song(int id, String name) {
        this.id = id;
        this.name = name;
    }

    public int getId() {
        return id;
    }

    public String getName() {
        return name;
    }
}

class User {
    private int id;
    private String name;

    public User(int id, String name) {
        this.id = id;
        this.name = name;
    }

    public int getId() {
        return id;
    }

    public String getName() {
        return name;
    }
}

class RecentlyPlayedStore {
    private int capacity;
    private Map<User, LinkedList<Song>> store;

    public RecentlyPlayedStore(int capacity) {
        this.capacity = capacity;
        this.store = new HashMap<>();
    }

    public void addSong(User user, Song song) {
        if (!store.containsKey(user)) {
            store.put(user, new LinkedList<>());
        }
        LinkedList<Song> playlist = store.get(user);
        playlist.addFirst(song);
        if (playlist.size() > capacity) {
            playlist.removeLast();
        }
    }

    public List<Song> getRecentlyPlayed(User user) {
        if (!store.containsKey(user)) {
            return Collections.emptyList();
        }
        return store.get(user);
    }
}
